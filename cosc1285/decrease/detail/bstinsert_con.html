  bool BSTNode::add(int value) {
      if (value == this->value)
            return false;
      else if (value < this->value) {
            if (left == NULL) {
                  left = new BSTNode(value);
                  return true;
            } else
                  return left->add(value);
      } else if (value > this->value) {
            if (right == NULL) {
                  right = new BSTNode(value);
                  return true;
            } else
                  return right->add(value);
      }
      return false;
}
